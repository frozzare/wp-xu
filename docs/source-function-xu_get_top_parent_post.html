 52: function xu_get_top_parent_post( $post = null ) {
 53:     if ( is_null( $post ) ) {
 54:         $post = get_post();
 55:     }
 56: 
 57:     if ( is_numeric( $post ) ) {
 58:         $post = get_post( (int) $post );
 59:     }
